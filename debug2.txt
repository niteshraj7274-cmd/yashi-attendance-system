    16	function getDistanceFromLatLonInM(lat1: number, lon1: number, lat2: number, lon2: number) {
    17	  var R = 6371;
    18	  var dLat = deg2rad(lat2-lat1);
    19	  var dLon = deg2rad(lon2-lon1); 
    20	  var a = 
    21	    Math.sin(dLat/2) * Math.sin(dLat/2) +
    22	    Math.cos(deg2rad(lat1)) * Math.cos(deg2rad(lat2)) * 
    23	    Math.sin(dLon/2) * Math.sin(dLon/2)
    24	    ; 
    25	  var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); 
    26	  var d = R * c * 1000;
    27	  return Math.round(d);
    28	}

    30	function deg2rad(deg: number) {
    31	  return deg * (Math.PI/180)
    32	}
